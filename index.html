SQL — criar tabelas
Execute no SQL Editor do Supabase:
-- 1. lancamentos create table if not exists lancamentos ( id uuid primary key default gen_random_uuid(), tipo text not null check (tipo in ('entrada','saida')), descricao text not null, categoria text, valor numeric(12,2) not null, data date not null, mes text, observacao text, created_at timestamptz default now(), updated_at timestamptz default now() ); -- 2. contas_pagar create table if not exists contas_pagar ( id uuid primary key default gen_random_uuid(), descricao text not null, valor numeric(12,2) not null, vencimento date not null, status text default 'pendente', lembrete_wpp boolean default true, dias_antes int[] default '{1,0}', msg_personalizada text, created_at timestamptz default now(), updated_at timestamptz default now() ); -- 3. lembretes_log create table if not exists lembretes_log ( id uuid primary key default gen_random_uuid(), conta_id uuid references contas_pagar(id) on delete cascade, enviado_em timestamptz default now(), dias_antes int, sucesso boolean, mensagem text, erro text ); -- 4. configuracoes create table if not exists configuracoes ( chave text primary key, valor text, updated_at timestamptz default now() ); -- Ver o arquivo 01_supabase_setup.sql para o setup completo -- com funções RPC, triggers e políticas RLS.
